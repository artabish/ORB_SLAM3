cmake_minimum_required(VERSION 3.27.1)
list(APPEND VCPKG_FEATURE_FLAGS "-compilertracking,manifests,registries,versions")
list(REMOVE_DUPLICATES VCPKG_FEATURE_FLAGS)
list(APPEND VCPKG_INSTALL_OPTIONS "--x-abi-tools-use-exact-versions")
list(APPEND VCPKG_INSTALL_OPTIONS "--clean-after-build")
list(REMOVE_DUPLICATES VCPKG_INSTALL_OPTIONS)
set(CMAKE_TOOLCHAIN_FILE ${CMAKE_CURRENT_SOURCE_DIR}/vcpkg/scripts/buildsystems/vcpkg.cmake CACHE STRING "Vcpkg toolchain file")

project(ORB_SLAM3 LANGUAGES CXX)

IF(NOT CMAKE_BUILD_TYPE)
  SET(CMAKE_BUILD_TYPE Release)
ENDIF()

MESSAGE("Build type: " ${CMAKE_BUILD_TYPE})

set(CMAKE_CXX_STANDARD 20)
if (MSVC)
  add_compile_options(/bigobj)
else ()
  add_compile_options(-Wa,-mbig-obj)
endif ()
LIST(APPEND CMAKE_MODULE_PATH ${PROJECT_SOURCE_DIR}/cmake_modules)

find_package(OpenCV REQUIRED core imgproc highgui calib3d features2d)

MESSAGE("OPENCV VERSION:")
MESSAGE(${OpenCV_VERSION})
find_package(Boost REQUIRED COMPONENTS serialization)
find_package(Eigen3 REQUIRED)
find_package(OpenSSL REQUIRED)
find_package(Pangolin REQUIRED)
find_package(Sophus CONFIG REQUIRED)
find_package(g2o CONFIG REQUIRED)
add_subdirectory(DBow2)

include_directories(
${PROJECT_SOURCE_DIR}
${PROJECT_SOURCE_DIR}/include
${PROJECT_SOURCE_DIR}/include/CameraModels
${EIGEN3_INCLUDE_DIR}
${Pangolin_INCLUDE_DIRS}
)

set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${PROJECT_SOURCE_DIR}/lib)

add_library(${PROJECT_NAME} STATIC
src/System.cc

src/Config.cc
src/Settings.cc
include/System.h
include/Tracking.h

target_link_libraries(${PROJECT_NAME} PUBLIC
${OpenCV_LIBS}
${EIGEN3_LIBS}
${Pangolin_LIBRARIES}
${dbow2_LIBRARIES}
g2o::core g2o::stuff g2o::types_icp g2o::types_sba g2o::types_sim3

)
